Penting: Kode ini sudah diperbarui dengan fungsi doPost() untuk mendukung upload file dari dashboard ke Drive. Jika sebelumnya sudah deploy, buat New Deployment (bukan edit yang lama) agar URL tetap valid.
Kode Apps Script (copy semua):
// ══ BACA file dari folder Drive (READ) ══
function doGet(e) {
  var folderId = e.parameter.folderId;
  if (!folderId) return json({error:"folderId required"});
  try {
    var folder = DriveApp.getFolderById(folderId);
    var files = [];
    var iter = folder.getFiles();
    while (iter.hasNext()) {
      var f = iter.next();
      var name = f.getName();
      if (name.match(/\.(xlsx|xls)$/i)) {
        files.push({id: f.getId(), name: name, size: f.getSize(),
                    modified: f.getLastUpdated().toISOString()});
      }
    }
    return json({files: files});
  } catch(err) {
    return json({error: err.message});
  }
}

// ══ UPLOAD file ke folder Drive (WRITE) ══
function doPost(e) {
  try {
    // Terima payload dari fetch (postData.contents) ATAU dari form field (parameter.payload)
    var raw = '';
    if (e.postData && e.postData.contents) {
      raw = e.postData.contents;
    } else if (e.parameter && e.parameter.payload) {
      raw = e.parameter.payload;
    }
    if (!raw) return json({error: 'No payload received'});

    var params   = JSON.parse(raw);
    var folderId = params.folderId;
    var fileName = params.fileName;
    var fileData = params.fileData;   // base64
    var mimeType = params.mimeType || 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet';

    if (!folderId || !fileName || !fileData) return json({error:"folderId, fileName, fileData required"});

    var folder  = DriveApp.getFolderById(folderId);
    var decoded = Utilities.base64Decode(fileData);
    var blob    = Utilities.newBlob(decoded, mimeType, fileName);

    // Hapus file lama dengan nama sama (replace)
    var existing = folder.getFilesByName(fileName);
    while (existing.hasNext()) { existing.next().setTrashed(true); }

    var newFile = folder.createFile(blob);
    return json({success: true, fileId: newFile.getId(), fileName: newFile.getName()});
  } catch(err) {
    return json({error: err.message});
  }
}

function json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
